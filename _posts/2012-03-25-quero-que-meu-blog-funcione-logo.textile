Function.prototype.expose = function( private ){
    this.fn = this.fn || undefined;

    if ( this.fn ){
        return this.fn.get( private )
    };

    var get = function( v ){
        return eval( v );
    };

    var fnSource = this.toString().match( /function.+?\((.*)\).+?\{([\s\S]*)\}/ );
    var F = new Function( fnSource[1],'this.get='+get+';'+fnSource[2] );
    this.fn = new F;
    return this.fn.get(private);
}
